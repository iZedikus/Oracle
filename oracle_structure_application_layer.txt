│   │   │       │       │   │
│   │   │       │       │   └── HandleIncomingTransactionUseCase.java
│   │   │       │       │       Назначение: принять входящую транзакцию от Симулякра,
│   │   │       │       │         найти подходящий WatchProfile, запустить матчинг.
│   │   │       │       │       Command: IncomingTransactionCommand
│   │   │       │       │         { externalTransactionID, accountId, paymentToken,
│   │   │       │       │           amount, currency, creditDebitIndicator,
│   │   │       │       │           merchantName, merchantId, mccCode,
│   │   │       │       │           bookingDateTime, valueDateTime,
│   │   │       │       │           debtorName, creditorName }
│   │   │       │       │       Алгоритм:
│   │   │       │       │         1. Создать IncomingTransaction.receive(...)
│   │   │       │       │         2. Сохранить (IncomingTransactionRepository.save)
│   │   │       │       │         3. Найти WatchProfile по paymentToken/accountId
│   │   │       │       │         4. Если профиль не найден или не Active →
│   │   │       │       │              markUnmatched(), save, return
│   │   │       │       │         5. tx.markProcessing(), save
│   │   │       │       │         6. Для каждого ActiveRule (findActiveRules()):
│   │   │       │       │              attempt = MatchAttempt.attempt(tx, rule)
│   │   │       │       │              save attempt
│   │   │       │       │              if Matched → break
│   │   │       │       │         7. Если хотя бы одно Matched:
│   │   │       │       │              tx.markMatched()
│   │   │       │       │              triggerEvent = TriggerEvent.create(attempt, profile)
│   │   │       │       │              triggerEvent.scheduleDelivery()
│   │   │       │       │              save triggerEvent
│   │   │       │       │              TriggerEventSenderPort.send(triggerEvent)
│   │   │       │       │         8. Иначе: tx.markUnmatched()
│   │   │       │       │         9. save tx
│   │   │       │       │
│   │   │       │       ├── profile/
│   │   │       │       │   │
│   │   │       │       │   ├── RegisterWatchProfileUseCase.java
│   │   │       │       │   │   Назначение: создать новый WatchProfile при action=REGISTER.
│   │   │       │       │   │   Command: RegisterWatchProfileCommand
│   │   │       │       │   │     { externalUserID, externalUserScenarioID,
│   │   │       │       │   │       scenarioTypeCode, paymentToken, bankBic,
│   │   │       │       │   │       ruleVersion, rules: List<RuleConditionDto>,
│   │   │       │       │   │       debitConfig: DebitConfigDto }
│   │   │       │       │   │   Алгоритм:
│   │   │       │       │   │     1. Если профиль с данным externalUserScenarioID уже
│   │   │       │       │   │          существует → бросить DuplicateProfileException
│   │   │       │       │   │     2. Смапить rules → List<RuleCondition>
│   │   │       │       │   │     3. WatchProfile.create(...)
│   │   │       │       │   │     4. save(profile)
│   │   │       │       │   │
│   │   │       │       │   ├── UpdateWatchProfileRulesUseCase.java
│   │   │       │       │   │   Назначение: обновить правила при action=UPDATE_RULES.
│   │   │       │       │   │   Command: UpdateRulesCommand
│   │   │       │       │   │     { externalUserScenarioID, ruleVersion,
│   │   │       │       │   │       rules: List<RuleConditionDto> }
│   │   │       │       │   │   Алгоритм:
│   │   │       │       │   │     1. Найти профиль по externalUserScenarioID
│   │   │       │       │   │     2. Найти ActiveRule по externalUserScenarioID
│   │   │       │       │   │     3. rule.updateConditions(conditions, version)
│   │   │       │       │   │     4. save(profile)

│   │   │       │               Алгоритм:
│   │   │       │                 1. TriggerEventRepository.findPendingForRetry()

│   │   │       │                        → при успехе: event.markDelivered()
│   │   │       │                        → при ошибке: event.scheduleRetry()
│   │   │       │                 3. save(event) для каждого
